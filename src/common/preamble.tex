\newcommand{\AphorismBody}[2]{%
  \vspace*{\fill}
  \begin{center}
    \begin{varwidth}{0.8\textwidth}
      \centering
      {\large\itshape #1\par}
      \vspace{0.25\baselineskip}
      {\normalfont\small\raggedleft --- #2\par}
    \end{varwidth}
  \end{center}
  \vspace*{\fill}
}

\newcommand{\AphorismPair}[2]{%
  \vspace*{\fill}
  \begin{center}

    \end{tabular}\par}
  \end{center}
  \vspace*{\fill}
}

% --- Helpers ---
\newcommand{\VersoTriplet}[3]{%
  \thispagestyle{empty}%
  \vspace*{\fill}
  \begin{center}
    \begin{varwidth}{0.78\textwidth}
      \raggedright\small
      \setlength{\parskip}{0.16\baselineskip}
      \setlength{\parindent}{0pt}
      #1\par
      #2\par
      #3\par
    \end{varwidth}
  \end{center}
  \vspace*{\fill}
  \clearpage
}

% One record, two views:
% - Always: recto = work
% - Decoded edition: verso = 3-line guide
% - Default edition: verso = blank
\newcommand{\WorkEntry}[6]{% id, title, body, lens, link, question
  \clearpage
  \ifnotesedition
    \VersoTriplet{#4}{#5}{#6}
  \else
    \BlankVerso
  \fi
  \cleardoublepage
  \thispagestyle{empty}
  \begin{flushleft}\setlength{\parindent}{0pt}
  \ifthenelse{\equal{#2}{}}{}{\WorkTitle{#2}}
  #3
  \end{flushleft}
}
