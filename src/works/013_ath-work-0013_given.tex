\WorkEntry{ATH-WORK-0013}{Given}{%

\begin{poem}
\begin{stanza}
Love is not a debt.\\
It cannot be taken.\\
You thought you stole\\
All that I had.\\
But you hadn't,\\
It was freely given.
\end{stanza}

\begin{stanza}
Love is not earned,\\
It cannot be bartered.\\
You thought I was a fool,\\
Easily taken,\\
Yet all my goods,\\
Were freely given.
\end{stanza}

\begin{stanza}
You cannot take my love,\\
I'm proud to have given.\\
\end{stanza}
\begin{stanza}
And you?
\end{stanza}
\end{poem}

}{\VLabel{Lens:}  What was later mishandled may already have been freely given---love given cannot be converted into theft.}
{\VLabel{Link:}  Love / Theft.}
{\VLabel{Question:}  What do you possess, if what you thought you stole was freely given?}
