\WorkEntry{ATH-WORK-0012}{After}{%
\begin{poem}
\begin{stanza}
What will you pursue?\\
The innocence of youth?\\
Will you accept\\
an element of truth?\\
\end{stanza}
\begin{stanza}
What will you dream?\\
Of nights immemorial?\\
Will you accept\\
a fleeting memorial?\\
\end{stanza}
\begin{stanza}
Who will you hold\\
that needs you most?\\
Will you accept\\
your spectre and ghost?\\
\end{stanza}
\begin{stanza}
Where will you turn\\
when you are bereft?\\
Will you accept\\
there's no one left?\\
\end{stanza}
\begin{stanza}
Where is your destiny\\
when circumstances fall?\\
Will you accept\\
you're in charge of it all?
\end{stanza}
\end{poem}
}{\VLabel{Lens:}  Acceptance is command.}{\VLabel{Link:}  Ghost / Memorial / Choice.}{\VLabel{Question:}  When nothing remains---what remains yours?}
