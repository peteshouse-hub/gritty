\WorkEntry{ATH-WORK-0002}{Remnant}{%

\begin{poem}
\begin{stanza}
Sixteen thousand headstrong\\
A floating sea of mediocrity\\
Where is my mind?\\
It's over the sea\\
\end{stanza}
\begin{stanza}
Your wave, it's like all others\\
Destined to swell\\
Destined to rise\\
Destined to fall\\
\end{stanza}
\begin{stanza}
Where is my mind?\\
\end{stanza}
\begin{stanza}
Sixteen thousand faces\\
Combine to amplify banality\\
Sixteen thousand faces\\
With eyes that do not see\\
\end{stanza}
\begin{stanza}
Where is my mind?\\
\end{stanza}
\begin{stanza}
The waves alone, they beat no drum\\
The crowd itself, the weakest link\\
Poseidon stills the seas\\
As I return to ego and id
\end{stanza}
\end{poem}
}{\VLabel{Lens:}  Mirrored cycles of depression.}{\VLabel{Link:}  Introspection / Extrospection / Individuality / Return.}
{\VLabel{Question:}  Does your sense of self, of who you are, change in cycles?}
