\WorkEntry{ATH-WORK-0004}{Stay}{%
\begin{poem}
\begin{stanza}
I think I will open the door today\\
I'll lay on the floor, and wait for He\\
I'm sure this is what life is for\\
Simply open the fucking door\\
\end{stanza}
\begin{stanza}
Kids are at school, and that's a chore\\
To end it all, I'll open the door\\
There is no food for any plate\\
I failed them all, I will suffocate\\
\end{stanza}
\begin{stanza}
The gasometer bloated stands tall\\
And this is how I end it all\\
Gas mark seven, do not ignite\\
I'm going down without a fight\\
\end{stanza}
\begin{stanza}
Coal gas no more, I rise from the floor\\
My head removed from oven door\\
It's fossil gas, so I'm not done-in\\
Now where did I put that fucking gin?
\end{stanza}
\end{poem}
